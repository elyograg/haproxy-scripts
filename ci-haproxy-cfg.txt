global
	maxconn 4096
	daemon
	#debug
	#quiet
	spread-checks	2
	tune.bufsize	65536
	stats socket /etc/haproxy/stats.socket

defaults
	mode	http
	balance	leastconn
	option	httplog
	option	dontlognull
	option	redispatch
	retries	1
	compression algo gzip
	compression type text/css text/html text/javascript application/javascript text/plain text/xml application/json application/css
	timeout connect	5s
	timeout client	15s
	timeout server	120s
	timeout http-keep-alive	5s
	retry-on all-retryable-errors

frontend web80
	bind 0.0.0.0:80 name web80
	default_backend be-local-81

backend be-local-81
	no log
	option httpchk
	server localhost 127.0.0.1:81
